# Golden Gate Bridge 3D Portfolio - Implementation Guide

## Quick Start

1. Clone the repository:
```bash
git clone https://github.com/surendravattikuti01-coder/my-portfolio.git
cd my-portfolio
```

2. Install dependencies:
```bash
npm install
```

3. Start development server:
```bash
npm run dev
```

4. Open `http://localhost:5173` in your browser

## Key Components

### GoldenGate3D.tsx
This component renders the 3D Golden Gate Bridge using Three.js:

```tsx
import React, { useEffect, useRef } from 'react';
import * as THREE from 'three';
import gsap from 'gsap';

const GoldenGate3D: React.FC = () => {
  const containerRef = useRef<HTMLDivElement>(null);

    useEffect(() => {
        if (!containerRef.current) return;

            // Scene setup
                const scene = new THREE.Scene();
                    const camera = new THREE.PerspectiveCamera(
                          75,
                                window.innerWidth / window.innerHeight,
                                      0.1,
                                            10000
                                                );
                                                    const renderer = new THREE.WebGLRenderer({ antialias: true, alpha: true });
                                                        renderer.setSize(window.innerWidth, window.innerHeight);
                                                            renderer.setClearColor(0x0a0e27, 1);
                                                                containerRef.current.appendChild(renderer.domElement);

                                                                    // Lighting
                                                                        const sunLight = new THREE.DirectionalLight(0xff6b1a, 1.2);
                                                                            sunLight.position.set(300, 200, 200);
                                                                                scene.add(sunLight);

                                                                                    const ambientLight = new THREE.AmbientLight(0x404040, 0.8);
                                                                                        scene.add(ambientLight);

                                                                                            // Create Bridge Structure
                                                                                                const bridgeGroup = new THREE.Group();
                                                                                                    
                                                                                                        // Main cables
                                                                                                            const cableGeometry = new THREE.BufferGeometry();
                                                                                                                const cablePoints = [];
                                                                                                                    // Create cable curve points
                                                                                                                        for (let i = 0; i <= 100; i++) {
                                                                                                                              const x = (i / 100) * 2000 - 1000;
                                                                                                                                    const y = 300 + Math.sin(i / 100 * Math.PI) * 200;
                                                                                                                                          const z = 0;
                                                                                                                                                cablePoints.push(new THREE.Vector3(x, y, z));
                                                                                                                                                    }
                                                                                                                                                        cableGeometry.setFromPoints(cablePoints);
                                                                                                                                                            const cableMaterial = new THREE.LineBasicMaterial({ color: 0xff6b1a, linewidth: 3 });
                                                                                                                                                                const cables = new THREE.Line(cableGeometry, cableMaterial);
                                                                                                                                                                    bridgeGroup.add(cables);
                                                                                                                                                                    
                                                                                                                                                                        // Bridge deck
                                                                                                                                                                            const deckGeometry = new THREE.BoxGeometry(2000, 30, 100);
                                                                                                                                                                                const deckMaterial = new THREE.MeshStandardMaterial({
                                                                                                                                                                                      color: 0xff6b1a,
                                                                                                                                                                                            metalness: 0.3,
                                                                                                                                                                                                  roughness: 0.4
                                                                                                                                                                                                      });
                                                                                                                                                                                                          const deck = new THREE.Mesh(deckGeometry, deckMaterial);
                                                                                                                                                                                                              deck.position.y = -50;
                                                                                                                                                                                                                  bridgeGroup.add(deck);
                                                                                                                                                                                                                  
                                                                                                                                                                                                                      // Towers
                                                                                                                                                                                                                          for (let x of [-500, 500]) {
                                                                                                                                                                                                                                const towerGeometry = new THREE.BoxGeometry(50, 400, 50);
                                                                                                                                                                                                                                      const towerMaterial = new THREE.MeshStandardMaterial({
                                                                                                                                                                                                                                              color: 0xff6b1a,
                                                                                                                                                                                                                                                      metalness: 0.4,
                                                                                                                                                                                                                                                              roughness: 0.3
                                                                                                                                                                                                                                                                    });
                                                                                                                                                                                                                                                                          const tower = new THREE.Mesh(towerGeometry, towerMaterial);
                                                                                                                                                                                                                                                                                tower.position.set(x, 100, 0);
                                                                                                                                                                                                                                                                                      bridgeGroup.add(tower);
                                                                                                                                                                                                                                                                                          }
                                                                                                                                                                                                                                                                                          
                                                                                                                                                                                                                                                                                              scene.add(bridgeGroup);
                                                                                                                                                                                                                                                                                              
                                                                                                                                                                                                                                                                                                  // Water
                                                                                                                                                                                                                                                                                                      const waterGeometry = new THREE.PlaneGeometry(4000, 3000);
                                                                                                                                                                                                                                                                                                          const waterMaterial = new THREE.MeshStandardMaterial({
                                                                                                                                                                                                                                                                                                                color: 0x1a4d6d,
                                                                                                                                                                                                                                                                                                                      metalness: 0.5,
                                                                                                                                                                                                                                                                                                                            roughness: 0.3
                                                                                                                                                                                                                                                                                                                                });
                                                                                                                                                                                                                                                                                                                                    const water = new THREE.Mesh(waterGeometry, waterMaterial);
                                                                                                                                                                                                                                                                                                                                        water.rotation.x = -Math.PI / 2;
                                                                                                                                                                                                                                                                                                                                            water.position.y = -400;
                                                                                                                                                                                                                                                                                                                                                scene.add(water);
                                                                                                                                                                                                                                                                                                                                                
                                                                                                                                                                                                                                                                                                                                                    // Camera position - simulating driving across bridge
                                                                                                                                                                                                                                                                                                                                                        camera.position.set(0, 100, -1000);
                                                                                                                                                                                                                                                                                                                                                            camera.lookAt(0, 100, 0);
                                                                                                                                                                                                                                                                                                                                                            
                                                                                                                                                                                                                                                                                                                                                                // Animation loop
                                                                                                                                                                                                                                                                                                                                                                    const animate = () => {
                                                                                                                                                                                                                                                                                                                                                                          requestAnimationFrame(animate);
                                                                                                                                                                                                                                                                                                                                                                                
                                                                                                                                                                                                                                                                                                                                                                                      // Gentle wave animation
                                                                                                                                                                                                                                                                                                                                                                                            water.material.emissiveIntensity = 0.1 + Math.sin(Date.now() * 0.001) * 0.05;
                                                                                                                                                                                                                                                                                                                                                                                                  
                                                                                                                                                                                                                                                                                                                                                                                                        // Driving animation using GSAP
                                                                                                                                                                                                                                                                                                                                                                                                              gsap.to(camera.position, {
                                                                                                                                                                                                                                                                                                                                                                                                                      z: 1000,
                                                                                                                                                                                                                                                                                                                                                                                                                              duration: 20,
                                                                                                                                                                                                                                                                                                                                                                                                                                      repeat: -1,
                                                                                                                                                                                                                                                                                                                                                                                                                                              yoyo: true,
                                                                                                                                                                                                                                                                                                                                                                                                                                                      ease: 'sine.inOut'
                                                                                                                                                                                                                                                                                                                                                                                                                                                            });
                                                                                                                                                                                                                                                                                                                                                                                                                                                            
                                                                                                                                                                                                                                                                                                                                                                                                                                                                  renderer.render(scene, camera);
                                                                                                                                                                                                                                                                                                                                                                                                                                                                      };
                                                                                                                                                                                                                                                                                                                                                                                                                                                                          animate();
                                                                                                                                                                                                                                                                                                                                                                                                                                                                          
                                                                                                                                                                                                                                                                                                                                                                                                                                                                              // Handle window resize
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                  const handleResize = () => {
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                        camera.aspect = window.innerWidth / window.innerHeight;
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                              camera.updateProjectionMatrix();
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                    renderer.setSize(window.innerWidth, window.innerHeight);
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                        };
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                            window.addEventListener('resize', handleResize);
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                            
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                return () => {
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                      window.removeEventListener('resize', handleResize);
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                            containerRef.current?.removeChild(renderer.domElement);
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                };
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                  }, []);
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                  
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                    return <div ref={containerRef} style={{ width: '100%', height: '100vh' }} />;
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                    };
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                    
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                    export default GoldenGate3D;
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                    ```

                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                              ## Customization Tips
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                              
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                              ### Change Bridge Color
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                              Edit the `color` property in `GoldenGate3D.tsx`:
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                              ```tsx
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                              color: 0xff6b1a  // Change to any hex color
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                              ```
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                              
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                              ### Modify Camera Speed
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                              Adjust the `duration` in the GSAP animation:
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                              ```tsx
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                              duration: 20  // Lower = faster, Higher = slower
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                              ```
